-- ============================================================ -- SISTEMA DGT/INTERIOR — Base de datos ficticia educativa -- Motor: SQLite3 (compatible MySQL/PostgreSQL con mínimos cambios) -- ============================================================ CREATE TABLE ciudadanos ( id INTEGER PRIMARY KEY AUTOINCREMENT, dni TEXT NOT NULL UNIQUE, -- 8 dígitos + letra (00000001A → 99999999Z) nombre TEXT NOT NULL, apellido1 TEXT NOT NULL, apellido2 TEXT, sexo TEXT, -- 'Hombre' / 'Mujer' fecha_nac TEXT, -- ISO 8601: YYYY-MM-DD provincia_residencia TEXT, municipio TEXT, direccion TEXT, numero_via TEXT, piso TEXT, codigo_postal TEXT, telefono TEXT, telefono2 TEXT, email TEXT, nacionalidad TEXT DEFAULT 'Española', estado_civil TEXT, num_hijos INTEGER DEFAULT 0, profesion TEXT, situacion_laboral TEXT, nivel_estudios TEXT, ingresos_anuales INTEGER, banco_ref TEXT, num_cuenta TEXT, licencia_conducir TEXT DEFAULT 'No', tipo_licencia TEXT, puntos_carnet INTEGER DEFAULT 15, antecedentes_penales TEXT DEFAULT 'No', num_pasaporte TEXT, fecha_exp_dni TEXT, fecha_cad_dni TEXT, alerta_policial TEXT DEFAULT 'Sin alertas', notas_admin TEXT, created_at TEXT DEFAULT (datetime('now')), updated_at TEXT DEFAULT (datetime('now')) ); CREATE TABLE matriculas ( id INTEGER PRIMARY KEY AUTOINCREMENT, matricula TEXT NOT NULL UNIQUE, -- Formato: '0001 BBB' → '9999 NZZ' dni TEXT NOT NULL, tipo_vehiculo TEXT, marca TEXT, modelo TEXT, año_fab INTEGER, fecha_mat TEXT, estado TEXT DEFAULT 'Activa', combustible TEXT, cv INTEGER, co2_grkg INTEGER, bastidor TEXT UNIQUE, fecha_itv TEXT, color TEXT, uso TEXT, km INTEGER DEFAULT 0, num_puertas INTEGER, num_plazas INTEGER, tara_kg INTEGER, masa_max_kg INTEGER, seguro_compañia TEXT, seguro_poliza TEXT, seguro_hasta TEXT, num_infracciones INTEGER DEFAULT 0, importe_multas INTEGER DEFAULT 0, titular_anterior TEXT, notas_admin TEXT, created_at TEXT DEFAULT (datetime('now')), updated_at TEXT DEFAULT (datetime('now')), FOREIGN KEY (dni) REFERENCES ciudadanos(dni) ON UPDATE CASCADE ); CREATE TABLE alertas_policiales ( id INTEGER PRIMARY KEY AUTOINCREMENT, dni TEXT NOT NULL, tipo_alerta TEXT, -- Busca y Captura / Requisitoria Judicial / Vigilancia / Multas motivo TEXT, emisor TEXT, fecha_emision TEXT, fecha_caducidad TEXT, juzgado TEXT, num_expediente TEXT, peligrosidad TEXT DEFAULT 'Media', observaciones TEXT, activa INTEGER DEFAULT 1, notas_admin TEXT, created_at TEXT DEFAULT (datetime('now')), FOREIGN KEY (dni) REFERENCES ciudadanos(dni) ); -- Cómo calcular la letra del DNI: -- letra = 'TRWAGMYFPDXBNJZSQVHLCKE'[numero % 23] -- Ejemplo: DNI 12345678 → 12345678 % 23 = 14 → letra Z → 12345678Z -- Rango DNI: 00000001A → 99999999Z (100 millones de combinaciones) -- Rango matrícula: 0001 BBB → 9999 NZZ -- Letras válidas matrícula: B,C,D,F,G,H,J,K,L,M,N,P,R,S,T,V,W,X,Y,Z (20 letras) -- Total combinaciones: 9.000 × 20³ = 72.000.000 matrículas posibles
